Требования к генерации:
1) Строгая согласованность с характеристиками игрока
2) Строгая согласованность с сущностями и логикой игрового мира
3) Категорически запрещается упоминать сущности, которые отсутствуют в игре
4) Комментарии в YAML можно убрать
5) Диалоги в разделах dialogs должны содержать строго от 5 до 10 вхождений

Тебе необходимо сгенерировать ровно один целостный квест типа "Доставка". 
Квест необходимо сгенерировать в строго таком формате YAML. Ничего дополнительно кроме данного YAML генерировать нельзя:
```yaml
quest:
  parts:
    # часть 1: получение задания
    resource_to_deliver:
      dialogs:  # диалоги с персонажем, который дает задание на добычу ресурса
        - speaker: ""
          remark: ""
      character: # Имя персонажа, который дает задание на добычу ресурса
      resource:  # название ресурса, который нужно добыть и затем доставить
      amount:  # количество ресурса, который нужно добыть и затем доставить

    # часть 2: встреча с противником после добычи ресурса
    enemy_to_face:
      dialogs:  # диалоги с противниками
        - speaker: ""
          remark: ""
      enemy: "" # название противника, с которым игрок встретится
      amount: # количество противников, минимум 1

    # часть 3: доставка ресурсов
    destination:
      # диалог с персонажем, которому нужно доставить добытый ресурс
      dialogs:
        - speaker: ""
          remark: ""
      # имя персонажа, которому нужно доставить добытый ресурс
      # персонажи (resource_to_deliver.character) и (destination.character_to_deliver) обязательно должны отличаться
      character_to_deliver: "" 

  objective_description: "" # описание цели квеста для игрока, не должен содержать упоминание противника
  # награда за прохождение квеста
  # запрещается давать игроку предмет в качестве награды, 
  # если он у него уже есть в инвентаре или в броне
  reward:
    item_name: ""

  # дать развернутые комментарии по генерации данного квеста
  # надо дать комментарий на каждый элемент структуры квеста
  explanation:
    resource_to_deliver:
      dialogs: ""
      character: ""
      resource: ""
      amount: ""

    enemy_to_face:
      dialogs: ""
      enemy: ""
      amount: ""

    destination:
      dialogs: ""
      character_to_deliver: ""

```
